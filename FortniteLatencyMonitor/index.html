// LÓGICA BACKEND (Python)

server.py
async def measure_latency(websocket):
    while True:
        latency = ping('aws-sa-east-1')
        data = {
            "ping": int(latency * 1000),
            "status": get_status(latency)
        }
        await websocket.send(json.dumps(data))
        await asyncio.sleep(1)
